with incomes_and_withdraws as (
    select *
    from d9_income
    union all
    select
        log_id,
        acc_id,
        dt,
        -amnt
    from d9_withdraw
)
select distinct
    dt,
    acc_id,
    sum(amnt) over (
        partition by acc_id
        order by dt
    ) as balance
from incomes_and_withdraws
